FROM python:3.11-slim

WORKDIR /app

# git is required at runtime (clone, commit, push inside pipelines)
RUN apt-get update && apt-get install -y --no-install-recommends git \
    && rm -rf /var/lib/apt/lists/* \
    && git config --global user.email "agent@pipeline.local" \
    && git config --global user.name "AI Pipeline Agent"

# Install deps first (layer cache — only rebuilds when requirements.txt changes)
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# Copy application modules
# Layout: app.py, config.py, models.py, llm.py, dag.py, job_store.py,
#         git_utils.py, workspace.py, routes/
COPY app.py config.py models.py llm.py dag.py job_store.py git_utils.py workspace.py ./
COPY routes/ ./routes/

EXPOSE 80

CMD ["uvicorn", "app:app", "--host", "0.0.0.0", "--port", "80"]
